import os
import sys
import json
import shutil
import zipfile
import subprocess
import threading
import platform
import socket
import uuid
from datetime import datetime, timezone

# CONFIG (placeholders are filled by the builder; networking/exfil is intentionally not used)
_SET = {
    "target": "{{DELIVERY}}",  # unused (kept for builder compatibility)
    "token": "{{BOT_TOKEN}}",  # unused (kept for builder compatibility)
    "id": "{{CHAT_ID}}",       # unused (kept for builder compatibility)
    "hook": "{{WEBHOOK}}",     # unused (kept for builder compatibility)
    "dbg": {{DEBUG}},
    "files": {{SENSITIVE}},    # if True, include a small, user-owned demo folder (see below)
    "secure": {{ANTI_VM}},     # unused (kept for builder compatibility)
    "ping": {{PING}}           # unused (kept for builder compatibility)
}


class NaviRecovery:
    """
    Safe local-only "recovery" stub.
    Collects basic, non-sensitive diagnostics and saves them to a ZIP on disk.
    No tokens, no browser secrets, no uploads.
    """

    def __init__(self):
        self.res = []
        self.user = self._safe_username()
        self.host = socket.gethostname()
        self.run_id = uuid.uuid4().hex[:8]

        base_tmp = os.getenv("TEMP") or os.getcwd()
        self.tmp = os.path.join(base_tmp, f"navi_recovery_{self.run_id}")
        os.makedirs(self.tmp, exist_ok=True)

        self.sys_info = {
            "timestamp_utc": datetime.now(timezone.utc).isoformat(),
            "host": self.host,
            "user": self.user,
            "os": f"{platform.system()} {platform.release()} ({platform.architecture()[0]})",
            "python": sys.version.replace("\n", " "),
        }

    def _log(self, m: str):
        if _SET.get("dbg"):
            print(f"  [*] {m}")
        self.res.append(m)

    @staticmethod
    def _safe_username() -> str:
        for k in ("USERNAME", "USER"):
            v = os.environ.get(k)
            if v:
                return v
        try:
            return os.getlogin()
        except Exception:
            return "Unknown"

    def _write_text(self, rel_path: str, content: str):
        p = os.path.join(self.tmp, rel_path)
        os.makedirs(os.path.dirname(p), exist_ok=True)
        with open(p, "w", encoding="utf-8", errors="replace") as f:
            f.write(content)

    def collect_system(self):
        self._log("Collecting system information...")
        self._write_text("system.json", json.dumps(self.sys_info, indent=2))

        # Basic environment (trimmed)
        env = {k: v for k, v in os.environ.items() if k.upper() in {
            "COMPUTERNAME", "USERNAME", "USERPROFILE", "HOMEPATH", "TEMP", "TMP",
            "PATH", "PROCESSOR_IDENTIFIER", "NUMBER_OF_PROCESSORS",
        }}
        self._write_text("env.json", json.dumps(env, indent=2))

    def collect_network(self):
        self._log("Collecting network snapshot...")
        cmds = [
            ("ipconfig.txt", ["ipconfig", "/all"]),
            ("route.txt", ["route", "print"]),
            ("dns_cache.txt", ["ipconfig", "/displaydns"]),
            ("netstat.txt", ["netstat", "-ano"]),
        ]
        for out_name, cmd in cmds:
            try:
                p = subprocess.run(cmd, capture_output=True, text=True, timeout=15, shell=False)
                data = (p.stdout or "") + ("\n\n[stderr]\n" + p.stderr if p.stderr else "")
                self._write_text(os.path.join("network", out_name), data.strip() + "\n")
            except Exception as e:
                self._write_text(os.path.join("network", out_name), f"error: {e}\n")

    def collect_processes(self):
        self._log("Collecting process list...")
        try:
            p = subprocess.run(["tasklist", "/v"], capture_output=True, text=True, timeout=20, shell=False)
            data = (p.stdout or "") + ("\n\n[stderr]\n" + p.stderr if p.stderr else "")
            self._write_text("processes/tasklist.txt", data.strip() + "\n")
        except Exception as e:
            self._write_text("processes/tasklist.txt", f"error: {e}\n")

    def take_screenshot(self):
        self._log("Capturing screenshot (best-effort)...")
        # Optional dependency: Pillow. If missing, skip silently.
        try:
            from PIL import ImageGrab  # type: ignore
            ss = ImageGrab.grab()
            ss.save(os.path.join(self.tmp, "screenshot.png"))
        except Exception as e:
            self._write_text("screenshot_error.txt", f"Screenshot unavailable: {e}\n")

    def include_demo_files(self):
        """
        Optional: include a small, user-owned demo folder to showcase "recovery".
        For safety, this only zips a folder named 'navi-demo-recovery' on Desktop if it exists.
        """
        if not _SET.get("files"):
            return
        self._log("Including demo files (if present)...")
        try:
            desk = os.path.join(os.path.expanduser("~"), "Desktop")
            src = os.path.join(desk, "navi-demo-recovery")
            if not os.path.isdir(src):
                self._write_text("demo_files.txt", "No demo folder found. Create: Desktop/navi-demo-recovery\n")
                return
            dst = os.path.join(self.tmp, "demo_files")
            shutil.copytree(src, dst, dirs_exist_ok=True)
        except Exception as e:
            self._write_text("demo_files_error.txt", f"error: {e}\n")

    def finalize_zip(self) -> str:
        self._log("Creating ZIP archive...")
        out_dir = os.path.join(os.getcwd(), "output")
        os.makedirs(out_dir, exist_ok=True)
        zip_name = f"Navi_Recovery_{self.host}_{self.user}_{self.run_id}.zip"
        zip_path = os.path.join(out_dir, zip_name)

        with zipfile.ZipFile(zip_path, "w", compression=zipfile.ZIP_DEFLATED) as zf:
            for root, _, files in os.walk(self.tmp):
                for fn in files:
                    full = os.path.join(root, fn)
                    rel = os.path.relpath(full, self.tmp)
                    zf.write(full, rel)

        try:
            shutil.rmtree(self.tmp)
        except Exception:
            pass

        self._log(f"Saved: {zip_path}")
        return zip_path

    def run(self):
        self._log("Initializing Navi Recovery (local-only)...")
        tasks = [
            self.collect_system,
            self.collect_network,
            self.collect_processes,
            self.take_screenshot,
            self.include_demo_files,
        ]
        threads = []
        for t in tasks:
            th = threading.Thread(target=t, daemon=True)
            th.start()
            threads.append(th)
        for th in threads:
            th.join(timeout=25)
        self.finalize_zip()


if __name__ == "__main__":
    NaviRecovery().run()
